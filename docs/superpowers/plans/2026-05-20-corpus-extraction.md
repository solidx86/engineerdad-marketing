# Corpus Extraction Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Land nine prose-layer corpus artifacts (3 evergreen `.md` methodology+proof, 2 dated snapshots, 2 author templates, 1 recipe append, 1 JSON per-fund snapshot) plus seven data-layer files under `corpus/data/` and three build/refresh scripts that keep the data fresh without re-implementing math.

**Architecture:** Three layers. **Layer 1 — prose** (items 1–6, 8, 9, 10; item 7 deferred): hand-authored markdown plus the structured per-fund JSON. **Layer 2 — data** (items A–G): seven JSON files content-gen reads directly via the `Read` tool so it never fabricates arithmetic. **Layer 3 — scripts** (3): `scripts/refresh-funds-corpus.mjs` regenerates items 6, 8, 10 monthly from the latest Public Mutual xlsx; `scripts/harvest-epf-simulations.mjs` drives the live EPF Sustainability Calculator via Playwright to produce item G; `scripts/build-derivative-tables.mjs` computes items A + B from item 1's assumption block. After all writes land, `/ingest-corpus` regenerates the BM25 sidecar so content-gen can retrieve the BM25-indexed additions.

**Tech Stack:** Node 20+, ESM, vitest, `exceljs` (new dep), the Playwright MCP tools (existing in registry), `python3 -m http.server` for the harvester's local server, bash. No new Python deps.

**Reference design:** `docs/superpowers/specs/2026-05-20-corpus-extraction-design.html` — read this before starting. It carries the full per-artifact spec, the bilingual policy, the compliance discipline, and the scenario / grid tables this plan executes against.

> **Deferred — automated corpus validation.** An earlier draft of this plan had a "Phase 3 — Corpus-additions validator" that scanned the new `.md` files for frontmatter conformance, bilingual sections, and banned compliance phrases. That has been pulled out and filed as backlog item **E-023** in `TASKS.md` ("Corpus audit pass"). In this plan, those quality checks are done by a manual review pass in Phase 8 instead.

Create `scripts/build-derivative-tables.mjs`:

```js
#!/usr/bin/env node
import fs from "node:fs/promises";
import path from "node:path";
import { fileURLToPath } from "node:url";

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const REPO_ROOT = path.resolve(__dirname, "..");

// Canonical assumptions — keep in sync with corpus/courses/epf-sustainability-model.md
const ASSUMPTION_SOURCE = "corpus/courses/epf-sustainability-model.md";

const RATES = [0.06, 0.07, 0.08, 0.10];
const YEARS = [5, 10, 15, 20, 25, 30, 35];
const PMTS = [100, 200, 300, 500, 750, 1000, 1500, 2000];
const TARGET_FVS = [50_000, 100_000, 200_000, 300_000, 500_000, 750_000, 1_000_000, 1_500_000];

export function fvAnnuity(monthlyPmt, annualRate, years) {
  if (monthlyPmt === 0) return 0;
  const r = annualRate / 12;
  const n = years * 12;
  if (r === 0) return monthlyPmt * n;
  return monthlyPmt * ((Math.pow(1 + r, n) - 1) / r);
}

export function requiredPmt(targetFv, annualRate, years) {
  if (targetFv === 0) return 0;
  const r = annualRate / 12;
  const n = years * 12;
  if (r === 0) return targetFv / n;
  return targetFv * r / (Math.pow(1 + r, n) - 1);
}

export function buildCompoundingTable() {
  const rows = [];
  for (const rate of RATES) for (const years of YEARS) for (const pmt of PMTS) {
    rows.push({ rate, years, monthly_pmt: pmt, fv: Math.round(fvAnnuity(pmt, rate, years)) });
  }
  return {
    generated_at: new Date().toISOString(),
    assumption_source: ASSUMPTION_SOURCE,
    grid: { rates: RATES, years: YEARS, monthly_pmts: PMTS },
    rows,
  };
}

export function buildMonthlyContributionTable() {
  const rows = [];
  for (const target_fv of TARGET_FVS) for (const years of YEARS) for (const rate of RATES) {
    rows.push({ target_fv, years, rate, monthly_pmt: Math.round(requiredPmt(target_fv, rate, years)) });
  }
  return {
    generated_at: new Date().toISOString(),
    assumption_source: ASSUMPTION_SOURCE,
    grid: { target_fvs: TARGET_FVS, years: YEARS, rates: RATES },
    rows,
  };
}

const isMain = import.meta.url === `file://${process.argv[1]}`;
if (isMain) {
  const dataDir = path.join(REPO_ROOT, "corpus", "data");
  await fs.mkdir(dataDir, { recursive: true });
  await fs.writeFile(path.join(dataDir, "compounding-table.json"), JSON.stringify(buildCompoundingTable(), null, 2));
  await fs.writeFile(path.join(dataDir, "monthly-contribution-required.json"), JSON.stringify(buildMonthlyContributionTable(), null, 2));
  console.log("wrote compounding-table.json + monthly-contribution-required.json");
}
```

#!/usr/bin/env node
import fs from "node:fs/promises";
import path from "node:path";
import { spawn } from "node:child_process";
import { fileURLToPath } from "node:url";

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const REPO_ROOT = path.resolve(__dirname, "..");
const CALCULATOR_DIR = "/Users/solid/Code/engineerdad-site/tools/epf-sustainability-calculator";
const PORT = 5174;
const OUTPUT_PATH = path.join(REPO_ROOT, "corpus", "data", "epf-sustainability-simulations.json");

// Grid definition (mirrors spec)
const AGES = [25, 30, 35, 40, 45, 50, 55];
const SALARIES = [3000, 5000, 7500, 10000, 15000, 20000];
const BALANCE_BUCKETS = [0, 0.5, 1.0, 1.5]; // multiples of RIA for that age
const RETIRE_AGES = [55, 60, 65];
const LIFESTYLES = ["Basic", "Adequate", "Enhanced"];

async function loadRiaBenchmarks() {
  // Reads corpus/data/kwsp-ria-benchmarks.json — must exist before harvesting
  const p = path.join(REPO_ROOT, "corpus", "data", "kwsp-ria-benchmarks.json");
  const txt = await fs.readFile(p, "utf8").catch(() => {
    throw new Error("Item C (kwsp-ria-benchmarks.json) must be authored before running the harvester. See Phase 4 task for C.");
  });
  const obj = JSON.parse(txt);
  const map = new Map(obj.tiers.map((t) => [t.age, t.ria_balance_rm]));
  return (age) => {
    const ages = [...map.keys()].sort((a, b) => a - b);
    const fit = ages.filter((a) => a <= age).pop() ?? ages[0];
    return map.get(fit) ?? 0;
  };
}

function buildInputGrid(riaFor) {
  const tuples = [];
  for (const age of AGES) {
    const ria = riaFor(age);
    for (const salary of SALARIES) for (const bucket of BALANCE_BUCKETS)
      for (const retireAge of RETIRE_AGES) for (const lifestyle of LIFESTYLES) {
        if (retireAge <= age) continue;
        tuples.push({
          age, salary_rm: salary,
          current_epf_rm: Math.round(ria * bucket),
          retire_age: retireAge, lifestyle,
        });
      }
  }
  return tuples;
}

function startStaticServer() {
  return new Promise((resolve, reject) => {
    const proc = spawn("python3", ["-m", "http.server", String(PORT)], {
      cwd: CALCULATOR_DIR, stdio: ["ignore", "pipe", "pipe"],
    });
    setTimeout(() => resolve(proc), 1500);
    proc.on("error", reject);
  });
}

async function harvest() {
  const riaFor = await loadRiaBenchmarks();
  const tuples = buildInputGrid(riaFor);
  console.log(`grid size: ${tuples.length} tuples`);

  const server = await startStaticServer();
  try {
    // === Playwright MCP integration ===
    // The implementer fills this section using the mcp__plugin_playwright_playwright__* tools.
    // Strategy (preferred path):
    //   1. browser_navigate to `http://localhost:${PORT}/index.html`
    //   2. For each tuple: browser_evaluate(`window.simulator(${JSON.stringify(t)})`)
    //   3. Collect outputs into `simulations` array
    // Strategy (fallback path):
    //   1. browser_navigate as above
    //   2. For each tuple: browser_fill_form, click Calculate, browser_evaluate to read result DOM
    //
    // This file owns: grid construction, server lifecycle, JSON output. The Playwright
    // invocations are performed by the agent executing this task.

    const simulations = await runSimulationsViaPlaywright(tuples);

    await fs.writeFile(OUTPUT_PATH, JSON.stringify({
      calculator_version: "v1.3.2 beta · 2026-03-26",  // read from APP_VERSION constant
      harvested_at: new Date().toISOString(),
      grid_definition: { ages: AGES, salaries: SALARIES, balance_buckets: BALANCE_BUCKETS, retire_ages: RETIRE_AGES, lifestyles: LIFESTYLES },
      simulations,
    }, null, 2));
    console.log(`wrote ${OUTPUT_PATH} with ${simulations.length} simulations`);
  } finally {
    server.kill();
  }
}
